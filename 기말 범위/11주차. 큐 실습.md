**1. 배열로 만들어진 원형 큐**
```c
#include <stdio.h>
#include <stdlib.h>

typedef struct {
	int max;   //큐의 최대 용량
	int front; //프런트
	int rear;  //리어
	int *que;  //큐의 맨 앞 요소에 대한 포인터
	int num;   //현재의 요소 개수
} Queue;

int Initialize(Queue*q, int max) { //큐 초기화
	q->que = (int*)malloc(max * sizeof(int));

	for (int i = 0; i < max; i++) {
		q->que[i] = 0;
	}
	q->max = max;
	q->num = max;
	q->front = 0;
	q->rear = 0;
	return 0;
}

int IsEmpty(Queue* q) { //큐가 비어있는가?
	return (q->front == q->rear % q->num);
}

int IsFull(Queue* q) { //큐가 가득 찼는가?
	return (q->front == (q->rear + 1) % q->num);
}

void Print(Queue* q) { //모든 데이터 출력
	for (int i = 0; i < q->num; i++)
		printf(" %d", q->que[i]);
}

int Enque(Queue*q, int x) { //큐에 데이터를 인큐
	if (IsFull(q)) {
		printf("overflow");
		Print(q);
		return -1;
	}					
	else {
		q->rear = (q->rear + 1) % (q->num);
		q->que[q->rear] = x;
		return 0;
	}
}

int Deque(Queue*q) { //큐에서 원소를 삭제한 후 해당 배열 원소 값을 0으로 치환
	if (IsEmpty(q)) {
		printf("underflow");
		return -1;
	}
	else {
		q->front = (q->front + 1) % (q->num);
		q->que[q->front] = 0;
		return 0;
	}
}

void Terminate(Queue*q) { //큐 종료
	if (q->que != NULL)
		free(q->que);							
	q->max = q->num = q->front = q->rear = 0;
}

int main() {
	int q;
	scanf("%d", &q); //원형 큐의 크기 입력

	Queue Que;
	Initialize(&Que, q); //큐 초기화

	int n, i;
	scanf("%d", &n); //연산의 개수 입력
	getchar();

	for (i = 0; i < n; i++) {
		char ch;
		scanf("%c", &ch);

		int ErrorCheck = 0;
		if (ch == 'I') { //삽입 
			int num;
			scanf("%d", &num);
			ErrorCheck = Enque(&Que, num);
		}
		if (ch == 'D') { //삭제
			ErrorCheck = Deque(&Que);
		}
		if (ch == 'P') { //출력
			Print(&Que);
			printf("\n");
		}
		if (ErrorCheck == -1) break;
		getchar();
	}

	Terminate(&Que); //큐 종료
	return 0;
}
```

```C
#include <stdio.h>
#include <stdlib.h>

typedef struct deque {
	int data;
	struct deque* prev;
	struct deque* next;
}Deque;

typedef struct dequetype {
	Deque* front, * rear;
}DequeType;

DequeType* getnode() {
	DequeType* horse = (DequeType*)malloc(sizeof(DequeType));

	horse->front = NULL;
	horse->rear = NULL;
	return horse;
}

int isEmpty(DequeType* que) {
	if (que->front == NULL)
		return 1;
	else
		return 0;
}

void add_front(DequeType *que, int n) {
	Deque* horse = (Deque*)malloc(sizeof(Deque));
	horse->data = n;

	if (que->front == NULL) {
		que->front = horse;
		que->rear = horse;
		horse->next = NULL;
		horse->prev = NULL;
	}
	else {
		que->front->prev = horse;
		horse->prev = NULL;
		horse->next = que->front;
		que->front = horse;
	}
}

void add_rear(DequeType *que, int n) {
	Deque* horse = (Deque*)malloc(sizeof(Deque));
	horse->data = n;

	if (que->rear == 0) {
		que->front = horse;
		que->rear = horse;
		horse->next = NULL;
		horse->prev = NULL;
	}
	else {
		que->rear->next = horse;
		horse->prev = que->rear;
		horse->next = NULL;
		que->rear = horse;
	}
}

int delete_front(DequeType* que) {
	Deque* horse = que->front;
	int r;
	if (isEmpty(que)) {
		printf("underflow");
		return -1;

		r = horse->data;

			que->front = NULL;
			que->rear = NULL;
		}
		else {
			que->front = que->front->next;
			que->front->prev = NULL;

		free(horse);
		return r;
	}
}

			que->front = NULL;
			que->rear = NULL;

			que->rear = que->rear->prev;
			que->rear->next = NULL;
		}

		free(horse);
		return r;
	}
	return 0;
}

	Deque* horse;
	for (horse = que->front; horse != NULL; horse = horse->next) {
		printf(" %d", horse->data);

	printf("\n");
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct deque {
	int element;
	struct deque* prev;
	struct deque* next;
}Deque;

Deque* getnode() {
	Deque* horse = (Deque*)malloc(sizeof(Deque));
	horse->element = 0;
	horse->next = NULL;
	horse->prev = NULL;
	return horse;
}

void add_front(Deque* front, Deque* rear, int n) {
	Deque* horse1, * horse2;
	horse1 = getnode();
	horse1->element = n;
	if (front->next == NULL) {
		front->next = horse1;
		rear->next = horse1;
	}
	else {
		horse2 = front->next;
		front->next = horse1;
		horse1->next = horse2;
		horse2->prev = horse1;
	}
}

void add_rear(Deque* front, Deque* rear, int n) {
	Deque* horse1, * horse2;
	horse1 = getnode();
	horse1->element = n;
	if (rear->next == NULL) {
		front->next = horse1;
		rear->next = horse1;
	}
	else {
		horse2 = rear->next;
		horse2->next = horse1;
		horse1->prev = horse2;
		rear->next = horse1;
	}
}

int delete_front(Deque* front, Deque* rear) {
	Deque* horse;
	if (front->next == NULL) {
		printf("underflow");
		return -1;
	}
	horse = front->next;
	if (front->next == rear->next) {
		front->next = NULL;
		rear->next = NULL;
		free(horse);
	}
	else {
		horse->next->prev = NULL;
		front->next = horse->next;
		free(horse);
	}
	return 0;
}

int delete_rear(Deque* front, Deque* rear) {
	Deque* horse;
	if (rear->next == NULL) {
		printf("underflow");
		return -1;
	}
	horse = rear->next;
	if (front->next == rear->next) {
		front->next = NULL;
		rear->next = NULL;
		free(horse);
	}
	else {
		horse->prev->next = NULL;
		rear->next = horse->prev;
		free(horse);
	}
	return 0;
}

void print(Deque* front, Deque* rear) {
	Deque* horse;
	for (horse = front->next; horse != NULL; horse = horse->next) {
		printf(" %d", horse->element);
	}
	printf("\n");
}
```
