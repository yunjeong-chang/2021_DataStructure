#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
	struct Node* left;
	struct Node* right;
	int data;
}Node;

Node* getNode() {
	Node* a = (Node*)malloc(sizeof(Node));
	a->left = NULL;
	a->right = NULL;
	a->data = '\0';
	return a;
}

Node* insertchild(int data, Node* left, Node* right) {
	Node* root = getNode();
	root->data = data;
	root->right = right;
	root->left = left;

	return root;
}


int main() {
	Node* F[9];
	F[7] = getNode();
	F[7]->data = 130;
	F[8] = getNode();
	F[8]->data = 80;

	F[4] = getNode();
	F[4]->data = 70;
	F[5] = getNode();
	F[5]->data = 90;

	F[6] = insertchild(120, F[7], F[8]);
	F[2] = insertchild(30, F[4], F[5]);
	F[3] = insertchild(50, NULL, F[6]);
	F[1] = insertchild(20, F[2], F[3]);

	int n;
	scanf("%d", &n);

	if (n > 8 || n < 1)	{
		printf("-1");
	}
	else if (F[n]->left == NULL && F[n]->right == NULL)	{
		printf("%d", F[n]->data);
	}
	else if (F[n]->left == NULL) {
		printf("%d %d", F[n]->data, F[n]->right->data);
	}
	else if (F[n]->right == NULL) {
		printf("%d %d", F[n]->data, F[n]->left->data);
	}
	else {
		printf("%d %d %d", F[n]->data, F[n]->left->data, F[n]->right->data);
	}

	return 0;
}
