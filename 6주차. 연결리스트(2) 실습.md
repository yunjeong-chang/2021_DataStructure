**1. 다항식 덧셈**
```c
#include <stdio.h>
#include <stdlib.h> 

typedef struct polynomial { //다항식의 각 항은 하나의 노드, 각 노드엔 세 개의 필드 저장
    int coef; //항의 계수
    int exp;  //항의 차수
    struct polynomia* next; //다음 노드를 가리키는 링크
}Poly;

Poly* getpoly() {
    Poly *newnode = (Poly*)malloc(sizeof(Poly));
    newnode->next = NULL;
    return newnode;
}

void appendTerm(Poly* k, int c, int e) { //다항식의 마지막 항을 표현하는 노드 k에 계수 c와 차수 e로 이루어진 새 항 추가
    while (k->next != NULL) {
        k = k->next;
    }
    Poly* newnode = getpoly();
    newnode->coef = c;
    newnode->exp = e;
    k->next = newnode;
}

Poly* addPoly(Poly *y1, Poly *y2) {
    Poly* result = getpoly();
    Poly *p = y1->next;
    Poly *q = y2->next;
    int sum;

    while (p != NULL && q != NULL) {
        if (p->exp > q->exp) {
            appendTerm(result, p->coef, p->exp);
            p = p->next;
        }
        else if (p->exp < q->exp) {
            appendTerm(result, q->coef, q->exp);
            q = q->next;
        }
        else {
            sum = p->coef + q->coef;
            if (sum != 0)
                appendTerm(result, sum, p->exp);
            p = p->next;
            q = q->next;
        }
    }
    while (p != NULL) {
        appendTerm(result, p->coef, p->exp);
        p = p->next;
    }
    while (q != NULL) {
        appendTerm(result, q->coef, q->exp);
        q = q->next;
    }
    return result;
}

void print(Poly *y3) {
    Poly* p = y3->next;
    while (p != NULL) {
        printf(" %d", p->coef);
        printf(" %d", p->exp);
        p = p->next;
    }
}

int main() {
    int N1 = 0, N2 = 0;
    int coef = 0, exp = 0;
    Poly* y1 = getpoly();
    Poly* y2 = getpoly();
    Poly* y3 = getpoly();

    scanf("%d", &N1); //첫 번째 다항식의 항의 개수
    for (int i = 0; i < N1; i++) {
        scanf("%d", &coef); //각 항의 계수
        scanf("%d", &exp);  //각 항의 지수
        appendTerm(y1, coef, exp);
    }

    scanf("%d", &N2); //두 번째 다항식의 항의 개수
    for (int i = 0; i < N2; i++) {
        scanf("%d", &coef); //각 항의 계수
        scanf("%d", &exp);  //각 항의 지수
        appendTerm(y2, coef, exp);
    }

    y3 = addPoly(y1, y2); //두 다항식의 덧셈
    print(y3);

    return 0;
}
```
