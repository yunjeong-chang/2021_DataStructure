**1. 다항식 덧셈**
```c
#include <stdio.h>
#include <stdlib.h> 

typedef struct node {
	int coef; //항의 계수
	int exp; //항의 차수	
	struct node* next;
}Node;

Node* create_node() {
	Node* newnode = (Node*)malloc(sizeof(Node));
	newnode->coef = 0;
	newnode->exp = 0;
	newnode->next = NULL;
	return newnode;
}

void print_list(Node* head) { //이중연결리스트 원소 출력
	Node* horse = head->next;

	while (horse != NULL) {
		printf(" %d", horse->coef);
		printf(" %d", horse->exp);
		horse = horse->next;
	}
}

void appendTerm(Node* k, int c, int e) {
	while (k->next != NULL) {
		k = k->next;
	}
	Node* newnode = create_node();
	newnode->coef = c;
	newnode->exp = e;
	k->next = newnode;
}


Node* addPoly(Node* y1, Node* y2) {
	Node* result = create_node();
	result->next = NULL;
	Node* horse1 = y1->next;
	Node* horse2 = y2->next;

	while (horse1 != NULL && horse2 != NULL) {
		if (horse1->exp > horse2->exp) {
			appendTerm(result, horse1->coef, horse1->exp);
			horse1 = horse1->next;
		}
		else if (horse1->exp < horse2->exp) {
			appendTerm(result, horse2->coef, horse2->exp);
			horse2 = horse2->next;
		}
		else {
			int sum;
			sum = horse1->coef + horse2->coef;
			if (sum != 0)
				appendTerm(result, sum, horse1->exp);
			horse1 = horse1->next;
			horse2 = horse2->next;
		}
	}
	while (horse1 != NULL) {
		appendTerm(result, horse1->coef, horse1->exp);
		horse1 = horse1->next;
	}
	while (horse2 != NULL) {
		appendTerm(result, horse2->coef, horse2->exp);
		horse2 = horse2->next;
	}
	return result;
}

int main() {
	int n1, n2, c, e;
	Node* y1, * y2, * y3;
	y1 = create_node();
	y2 = create_node();
	y3 = create_node();

	scanf("%d", &n1);
	for (int i = 0; i < n1; i++) {
		scanf("%d %d", &c, &e);
		appendTerm(y1, c, e);
	}
	scanf("%d", &n2);
	for (int i = 0; i < n2; i++) {
		scanf("%d %d", &c, &e);
		appendTerm(y2, c, e);
	}
	y3 = addPoly(y1, y2);
	print_list(y3);
}
```
