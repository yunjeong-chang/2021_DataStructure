```c
#include <stdio.h>
int isPalindrome(char A[], int n, int cnt) {
    if (cnt >= n / 2 + 1) { 
        return 1;
    }

    else if (A[0] != A[n - cnt]) { //같은 문자가 아니면
        return 0;
    }
     
    else { //같은 문자이면
        return isPalindrome(A + 1, n - 1, cnt + 1); 
    }
}

int main() {
    char str[100];
    int cnt = 1;
    scanf("%[^\n]s", &str); 
    printf("%d", isPalindrome(&str, strlen(str), cnt)); 
    return 0;
}
```

```c
#include <stdio.h>

int checkSudoku(int SudokuGrid[][9], int row, int col, int num)
{
    int rowStart = (row / 3) * 3;
    int colStart = (col / 3) * 3;

    for (int i = 0; i < 9; i++) {
        if (SudokuGrid[row][i] == num) 
            return 0;
        if (SudokuGrid[i][col] == num) 
            return 0;
        if (SudokuGrid[rowStart + (i % 3)][colStart + (i / 3)] == num) 
            return 0;
    }
    return 1;
}

int solveSudoku(int SudokuGrid[][9], int row, int col) { //주된 문제 해결 함수
    if (row < 9 && col < 9) {
        if (SudokuGrid[row][col] != 0) {
            if ((col + 1) < 9) 
                return solveSudoku(SudokuGrid, row, col + 1);
            else if ((row + 1) < 9) 
                return solveSudoku(SudokuGrid, row + 1, 0);
            else 
                return 1;
        }

        else {
            for (int i = 0; i < 9; i++) {
                if (checkSudoku(SudokuGrid, row, col, i + 1)) {
                    SudokuGrid[row][col] = i + 1;
                    if ((col + 1) < 9) {
                        if (solveSudoku(SudokuGrid, row, col + 1)) 
                            return 1;
                        else SudokuGrid[row][col] = 0;
                    }
                    else if ((row + 1) < 9) {
                        if (solveSudoku(SudokuGrid, row + 1, 0)) 
                            return 1;
                        else SudokuGrid[row][col] = 0;
                    }
                    else 
                        return 1;
                }
            }
        }

        return 0;
    }

    else 
        return 1;
}

int main()
{
    int SudokuGrid[9][9], i, j;

    for (i = 0; i < 9; i++) {
        for (j = 0; j < 9; j++) {
            scanf("%d", &SudokuGrid[i][j]); //처음 상태 스도쿠판 입력
        }
    }

    if (solveSudoku(SudokuGrid, 0, 0) == 1) { //조건 완성되면 최종 스도쿠판 출력
        for (i = 0; i < 9; i++) {
            for (j = 0; j < 9; j++) {
                printf("%d ", SudokuGrid[i][j]);
            }
            printf("\n");
        }
    }
    return 0;
}
```
