		return count;

		return 0;
}

		return 1; //무조건 부분집합이다
	else if (check_one(A) > 0)
		count = check_one(A);
	
	while (i != 26) {
		if (A[i] == 1 && B[i] == 0) 
			return 0;

			cnt++;
			i++;
			j++;
		}
		else {
			i++;
			j++;
		}
	}
	if (cnt == count)
		return 1;
	else
		return 0;
