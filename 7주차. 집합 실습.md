**1. 집합 A가 집합 B의 부분집합인지 검사 (헤더 없애야 함 이건 헤더 있는 버전)**
```c
#include <stdio.h>
#include <stdlib.h> 

typedef struct node {
	int data;
	struct node* next;
}Node;

Node* create_node() {
	Node* newnode = (Node*)malloc(sizeof(Node));
	newnode->data = 0;
	newnode->next = NULL;
	return newnode;
}

void addset(Node* s, int e) {
	Node* newnode = create_node();
	newnode->data = e;

	while (s->next != NULL) {
		s = s->next;
	}
	s->next = newnode;
}

int subset(Node* s1, Node* s2) {
	s1 = s1->next;
	s2 = s2->next;
	while (s1 != NULL && s2 != NULL) {
		if (s1->data > s2->data) {
			s2 = s2->next;
		}
		else if (s1->data == s2->data) {
			s1 = s1->next;
			s2 = s2->next;
		}
		else
			break;
	}
	if (s1 == NULL)
		printf("0");
	else
		printf("%d", s1->data);
}

int main() {
	int n1, n2, a, b;
	Node* A = create_node();
	Node* B = create_node();

	scanf("%d", &n1);
	for (int i = 0; i < n1; i++) {
		scanf("%d", &a);
		addset(A, a);
	}
	scanf("%d", &n2);
	for (int i = 0; i < n2; i++) {
		scanf("%d", &b);
		addset(B, b);
	}
	subset(A, B);
}
```

**2. 합집합과 교집합 구하기**
```c
#include <stdio.h>
#include <stdlib.h> 

typedef struct node {
	int data;
	struct node* next;
}Node;

Node* create_node() {
	Node* newnode = (Node*)malloc(sizeof(Node));
	newnode->data = 0;
	newnode->next = NULL;
	return newnode;
}

void print_list(Node* head) { //원소 출력
	Node* horse = head->next;

	while (horse != NULL) {
		printf(" %d", horse->data);
		horse = horse->next;
	}
}

void addset(Node* s, int e) {
	Node* newnode = create_node();
	newnode->data = e;

	while (s->next != NULL) {
		s = s->next;
	}
	s->next = newnode;
}

Node* union_set(Node* s1, Node* s2) {
	Node* newnode = create_node();
	s1 = s1->next;
	s2 = s2->next;

	if (s1 == NULL && s2 == NULL) {
		addset(newnode, 0);
		return newnode;
	}

	while (s1 != NULL || s2 != NULL) {
		if (s1 == NULL) {
			addset(newnode, s2->data);
			s2 = s2->next;
		}
		else if (s2 == NULL) {
			addset(newnode, s1->data);
			s1 = s1->next;
		}
		else {
			if (s1->data < s2->data) {
				addset(newnode, s1->data);
				s1 = s1->next;
			}
			else if (s1->data > s2->data) {
				addset(newnode, s2->data);
				s2 = s2->next;
			}
			else {
				addset(newnode, s1->data);
				s1 = s1->next;
				s2 = s2->next;
			}
		}
	}
	return newnode;
}

Node* intersect_set(Node* s1, Node* s2) {
	Node* newnode = create_node();
	s1 = s1->next;
	s2 = s2->next;

	if (s1 == NULL || s2 == NULL) {
		addset(newnode, 0);
		return newnode;
	}
	while (s1 != NULL && s2 != NULL) {
		if (s1->data < s2->data) {
			s1 = s1->next;
		}
		else if (s1->data > s2->data) {
			s2 = s2->next;
		}
		else {
			addset(newnode, s1->data);
			s1 = s1->next;
			s2 = s2->next;
		}
	}
	return newnode;
}

int main() {
	int n1, n2, a, b;
	Node* A = create_node();
	Node* B = create_node();

	scanf("%d", &n1);
	for (int i = 0; i < n1; i++) {
		scanf("%d", &a);
		addset(A, a);
	}
	scanf("%d", &n2);
	for (int i = 0; i < n2; i++) {
		scanf("%d", &b);
		addset(B, b);
	}

	print_list(union_set(A, B));
	printf("\n");
	print_list(intersect_set(A, B));
	return 0;
}
```
