**1번 - 이중연결리스트 이용하여 영문자 리스트 ADT 구현**
```c
#include <stdio.h>
#include <stdlib.h> 

typedef struct node {  //이중연결리스트의 각 노드는
    char data; //원소 및
    struct node* prev; //이전 노드를 가리키는 링크와
    struct node* next; //다음 노드를 가리키는 링크를 저장한다.
}Node;

Node* create_node() { //이중연결리스트 노드 생성
    Node* newnode = (Node*)malloc(sizeof(Node));
    newnode->data = 0;
    newnode->next = NULL;
    newnode->prev = NULL;
    return newnode;
}

void initialize(Node* header, Node* trailer) { //이중연결리스트로 초기화
    header->next = trailer;
    trailer->prev = header;
}

int count_node(Node* head) { //이중연결리스트 길이 측정
    Node* horse = head->next;
    int count = 0;
    while ((horse->next) != NULL) {
        count++;
        horse = horse->next;
    }
    return count;
}

void add_node(Node* head, int r, char e) { //이중연결리스트의 r번째에 e추가
    Node* horse = head;
    Node* newnode = create_node();
    if (r < 1 || r > count_node(head) + 1) {
        printf("invalid position\n");
        return;
    }

    for (int i = 1; i <= r; i++)
        horse = horse->next;

    newnode->data = e;
    newnode->next = horse;
    newnode->prev = horse->prev;
    horse->prev->next = newnode;
    horse->prev = newnode;
}

void delete_node(Node* head, int r) { //이중연결리스트에서 r번째 삭제
    Node* horse = head;
    if (r < 1 || r > count_node(head)) {
        printf("invalid position\n");
        return;
    }

    for (int i = 1; i <= r; i++)
        horse = horse->next;
  
    horse->prev->next = horse->next;
    horse->next->prev = horse->prev;
}

char get_node(Node* head, int r) { //이중연결리스트의 r번째 원소 반환
    Node* horse = head;
    if (r < 1 || r > count_node(head)) {
        printf("invalid position\n");
        return -1;
    }
    for (int i = 1; i <= r; i++)
        horse = horse->next;
    
    return horse->data;
}

void print_list(Node* head) { //이중연결리스트 원소 출력
    Node* horse = head->next;
    if (horse->data == 0) {
        printf("invalid position\n");
        return;
    }
    while (horse->data != NULL) {
        printf("%c", horse->data);
        horse = horse->next;
    }
}

int main() {
    int n, rank;
    char check, element, returnelem;
    Node* header, * trailer;

    header = create_node();
    trailer = create_node();
    initialize(header, trailer);

    scanf("%d", &n); //연산의 개수 입력
    getchar();

    while (n != 0) {
        scanf("%c", &check); //연산의 종류 입력(A, D, G, P 中 하나)

        if (check == 'A') { //원소 추가
            scanf("%d %c", &rank, &element);
            add_node(header, rank, element);
        }

        else if (check == 'D') { //원소 삭제
            scanf("%d", &rank);
            delete_node(header, rank);
        }

        else if (check == 'G') { //원소 반환
            scanf("%d", &rank);
            returnelem = get_node(header, rank);
            if (returnelem != -1)
                printf("%c\n", returnelem);
        }

        else if (check == 'P') { //모든 원소 출력
            print_list(header);
            printf("\n");
        }

        else {
            printf("invalid command\n");
            break;
        }

        getchar();
        n--;
    }

    return 0;
}
```

```c
void swap_node(Node* head, int r1, int r2) {
    Node* horse1 = head;
    Node* horse2 = head;
    if (r1 < 1 || r1 > count_node(head) || r2 < 1 || r2 > count_node(head)) {
        printf("invalid position\n");
        return ;
    }
    for (int i = 1; i <= r1; i++)
        horse1 = horse1->next;
    for (int i = 1; i <= r2; i++)
        horse2 = horse2->next;
    Node* tmp;
    tmp = horse1->data;
    horse1->data = horse2->data;
    horse2->data = tmp;
} 

void removeAll_node(Node* head) {
    Node* horse = head;
    Node* tmp;
    while (horse->next != NULL) {
        tmp = horse->next;
        free(horse);
        horse = tmp;
    }
}

int main() {
    int n, rank;
    char check, element, returnelem;
    Node* header, * trailer;

    header = create_node();

            print_list(header);
            printf("\n");
        }
        
        else if (check == 'R') {
            removeAll_node(header);
            header = create_node();
            trailer = create_node();
            initialize(header, trailer);
            printf("\n");
        }

        else if (check == 'C') {
            printf("%d\n", count_node(header));
            printf("\n");
        }

        else if (check == 'S') {
            int r1, r2;
            scanf("%d %d", &r1, &r2);
            swap_node(header, r1, r2);
        }

        else {
            printf("invalid command\n");
            break;
        }

        getchar();
        n--;
    }

    return 0;
}
```
