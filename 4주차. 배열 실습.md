```c
#include <stdio.h>

void Reverse(int x[], int n, int cnt, int info[]) {
    int start, end, tmp = 0;
    for (int i = 0; i < cnt * 2; i+= 2) {
         start = info[i];
         end = info[i + 1];

         for (int j = info[i]; j <= (info[i] + ((info[i + 1] - info[i]) / 2)); j++) {
             tmp = x[start];
             x[start] = x[end];
             x[end] = tmp;
             start++;
             end--;
         }
    }


}
int main() {
    int N, i, cnt;
    int X[100], info[200];

    scanf("%d", &N); //수열의 길이 N
    for (i = 0; i < N; i++)
        scanf("%d", &X[i]); //수열 X

    scanf("%d", &cnt); //뒤집기 정보의 개수
    for (i = 0; i < cnt * 2; i++)
        scanf("%d", &info[i]); //뒤집기 구간 정보

    Reverse(X, N, cnt, info);
    for (i = 0; i < N; i++)
        printf(" %d", X[i]);

    return 0;
}
```

2번 - 위치 변경 프로그램
```c
#include <stdio.h>

void IndexChange(int x[], int n, int cnt, int info[]) {
    int tmp[100], i = 0;

    tmp[info[i + 1]] = x[info[i + 1]];
    x[info[i + 1]] = x[info[i]];

    for (i = 2; i < cnt; i++) {
        tmp[info[i]] = x[info[i]];
        x[info[i]] = tmp[info[i - 1]];
    }


}
int main() {
    int N, i, cnt;
    int X[100], info[100];

    scanf("%d", &N); //수열의 길이 N
    for (i = 0; i < N; i++) {
        scanf("%d", &X[i]); //수열 X
    }
    scanf("%d", &cnt); //순서 바꿈 정보의 길이
    for (i = 0; i < cnt; i++) {
        scanf("%d", &info[i]); //순서 바꿈 정보
    }
    IndexChange(X, N, cnt, info);
    for (i = 0; i < N; i++) {
        printf(" %d", X[i]);
    }
    return 0;
}
```

3번 - 지그재그 배열
```c
#include <stdio.h>

int zigzag(int x[][100], int n) {
    int k = 1;
    for (int i = 0; i < n; i++) {
        if (i % 2 == 0) {
            for (int j = 0; j < n; j++) {
                x[i][j] = k;
                k++;
            }
        }
        else {
            for (int j = n - 1; j >= 0; j--) {
                x[i][j] = k;
                k++;
            }
        }
    }
}
int main() {
    int n, x[100][100];
    scanf("%d", &n); //행렬 크기 n 입력 -> n x n 행렬
    zigzag(x, n);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            printf(" %d", x[i][j]);
        }
        printf("\n");
    }
    return 0;
}
```

```c
#include <stdio.h>

int zigzag(int n) {
    int k = 1;
    for (int i = 0; i < n; i++) {
        if (i == 0) { //첫행
            for (int j = 0; j < n; j++)
                printf(" %d", k++);
        }
        else if (i % 2 == 0) { // 짝수행
            k += n + 1;
            for (int j = 0; j < n; j++)
                printf(" %d", k++);
        }
        else { //홀수행
            k += n - 1;
            for (int j = 0; j < n; j++)
                printf(" %d", k--);
        }
        printf("\n");
    }
}
int main() {
    int n;
    scanf("%d", &n); //행렬 크기 n 입력 -> n x n 행렬
    zigzag(n);
    return 0;
}
```

4번 - 나선형 배열, 달팽이 배열
```c
#include <stdio.h>

void spiral(int x[][100], int n, int m) {
    int k = 1; // 배열을 채워 나갈 값
    int col = 0, row = -1;
    int idx = 1;
    int end = n * m;

    while (1) {
        for (int i = 0; i < m; i++) {
            row += idx;
            x[col][row] = k++;
        }
        m--;

        for (int i = 0; i < n - 1; i++) {
            col += idx;
            x[col][row] = k++;
        }
        n--;

        idx *= -1;
        if (k == end + 1) break;
    }
}

int main() {
    int n, m, x[100][100];
    scanf("%d %d", &n, &m); //n x m 행렬 // 4 x 5
    spiral(x, n, m);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            printf(" %d", x[i][j]);
        }
        printf("\n");
    }
    return 0;
}
```

5번 - 대각선 출력
```c
#include <stdio.h>

void diagonal(int x[][100], int n, int m) {
    int k = 1; //배열에 채워 넣을 값
    int i = 0, j = 0;
    int row = 0, col = 0;

    for (i = 0; i < m; i++) { //우선 주대각선 전까지
        row = j, col = i;
        while (row < n && col >= 0) {
            x[row++][col--] = k++;
        }
    }
    i = m - 1;
    for (j = 1; j < n; j++) { //주대각선 이후
        row = j, col = i;
        while (row < n && col >= 0) {
            x[row++][col--] = k++;
        }
    }
}

int main() {
    int n, m, x[100][100];
    scanf("%d %d", &n, &m); //n x m 행렬 // 4 x 5
    diagonal(x, n, m);

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            printf(" %d", x[i][j]);
        }
        printf("\n");
    }
    return 0;
}
```
